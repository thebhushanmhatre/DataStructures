class CircularQueue {
  constructor(size) {

    this.queue = [];
    this.read = 0;
    this.write = 0;
    this.max = size - 1;

    while (size > 0) {
      this.queue.push(null);
      size--;
    }
  }

  print() {
    return this.queue;
  }

  enqueue(item) {
    // Only change code below this line
    if (this.queue[this.write]) {
      // we cannot overwrite
      return null;
    }

    this.queue[this.write] = item;
    if (this.write == this.max) {
      this.write = 0;
    } else {
      this.write += 1;
    }
    return item;
    // Only change code above this line
  }

  dequeue() {
    // Only change code below this line
    if (!this.queue[this.read]) {
      // there is no item to dequeue
      return null;
    }

    let itemToReturn = this.queue[this.read];

    this.queue[this.read] = null; // remove the item by overriding it to null
    if (this.read == this.max) {
      this.read = 0;
    } else {
      this.read += 1;
    }

    return itemToReturn;
    // Only change code above this line
  }
}
