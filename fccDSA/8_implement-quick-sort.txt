// Quick sort is an efficient, recursive divide-and-conquer approach to sorting an array. In this method, a pivot value is chosen in the original array. The array is then partitioned into two subarrays of values less than and greater than the pivot value. We then combine the result of recursively calling the quick sort algorithm on both sub-arrays. This continues until the base case of an empty or single-item array is reached, which we return. The unwinding of the recursive calls return us the sorted array.

// Quick sort is a very efficient sorting method, providing O(nlog(n)) performance on average. It is also relatively easy to implement. These attributes make it a popular and useful sorting method.

function quickSort(arr) {
  // Only change code below this line
  qs(arr, 0, arr.length - 1);
  return arr;
  // Only change code above this line
}

function qs(arr, start, end) {
  if (start < end) {
    let partitionIndex = partition(arr, start, end);
    qs(arr, start, partitionIndex - 1);
    qs(arr, partitionIndex + 1, end);
  }
}

function partition(arr, start, end) {
  // get a random pivot, do not use this for other than comparison
  // pivotIndex must be inside the current subarray range [start, end]
  let pivotIndex = Math.floor(Math.random()*(end-start+1)) + start; // don't forget (random + start)
  let pivotElement = arr[pivotIndex];

  // BUG FIX:
  // Move pivot to the end so partitioning works correctly
  [arr[pivotIndex], arr[end]] = [arr[end], arr[pivotIndex]];

  // find a partitionIndex 
  // such that elements to its left are lesser and to the right are greater
  let partitionIndex = start;
  for (let i = start; i < end; i++) {
    if (arr[i] <= pivotElement) {
      // swap
      [arr[i], arr[partitionIndex]] = [arr[partitionIndex], arr[i]];
      // increment as we find element greater than pivot
      partitionIndex++;
    }
  }

  // BUG FIX:
  // Place pivot in its correct sorted position
  [arr[partitionIndex], arr[end]] = [arr[end], arr[partitionIndex]];

  return partitionIndex;
}
